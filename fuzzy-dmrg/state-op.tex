\begin{tikzpicture}[>=Stealth, font=\sffamily]

    % =========================================================
    % DEFINITIONS & COLORS
    % =========================================================
    \colorlet{taucolor}{blue!60!black}  % Constant Time
    \colorlet{sigmacolor}{red!60!black} % Constant Space
    \colorlet{statecolor}{cyan!80!blue} % State Cross-section
    \colorlet{opcolor}{orange!90!red}   % Operator/Vacuum

    % Cylinder Parameters
    \def\Rx{1.5}     % X Radius
    \def\Ry{0.4}     % Y Radius (Perspective)
    \def\H{5}        % Height of cylinder
    \def\statetau{3} % Y-level (time) of the highlighted state
    
    % Plane Parameters
    \def\C{10}       % Center X of plane
    \def\Cy{2.5}     % Center Y of plane
    \def\Pr{3.2}     % Max radius on plane

    % =========================================================
    % LEGEND
    % =========================================================
    % \node[draw, rounded corners, fill=gray!5, inner sep=2ex, text=black] at (5, \H+1.5) {
    %     \begin{tabular}{cl}
    %     \tikz[baseline=-0.5ex]{\draw[taucolor, thick] (0,0) -- (0.5,0);} & Equal time $\tau$ / Constant radius $|z| = R$ \\[1ex]
    %     \tikz[baseline=-0.5ex]{\draw[sigmacolor, thick] (0,0) -- (0.5,0);} & Equal space $\sigma$ / Constant angle $\theta$
    %     \end{tabular}
    % };


    % =========================================================
    % LEFT: CYLINDER (w = tau + i*sigma)
    % =========================================================
    
    % Vacuum limit Arrow (Bottom Left)
    \draw[-Stealth, thick, opcolor] (-\Rx-2.5, 0) node[left, align=center, text=black] {$\tau \to -\infty$\\ ($|\mathcal O\rangle$)} to[out=0, in=180] (-\Rx-0.1, 0);

    % Axis Titles
    %\node[align=center, font=\large] at (0, -1.8) {\textbf{Cylinder ($\mathbb{R} \times S^1$)}\\ $w = \tau + i\sigma$};

    % --- Back Half of Cylinder ---
    % Constant Time (Tau) Lines - Back
    \foreach \y in {0, 1, 2, 4} {
        \draw[taucolor, dashed, line width=0.6pt] (\Rx, \y) arc[start angle=0, end angle=180, x radius=\Rx, y radius=\Ry];
    }
    % Constant Space (Sigma) Lines - Back
    \foreach \ang in {15, 45, 75, 105, 135, 165} {
        \draw[sigmacolor, dashed, line width=0.6pt] ({\Rx*cos(\ang)}, {\Ry*sin(\ang)}) -- ({\Rx*cos(\ang)}, {\H+\Ry*sin(\ang)});
    }
    
    % The State (Specific constant time slice) - Back
    %\draw[statecolor, dashed, line width=1.5pt] (\Rx, \statetau) arc[start angle=0, end angle=180, x radius=\Rx, y radius=\Ry];

    % --- Cylinder Shading (Creates 3D Surface) ---
    \shade[left color=gray!30, middle color=white, right color=gray!20, opacity=0.85] 
        (-\Rx, 0) arc[start angle=180, end angle=360, x radius=\Rx, y radius=\Ry] -- 
        (\Rx, \H) arc[start angle=360, end angle=180, x radius=\Rx, y radius=\Ry] -- cycle;

    % --- Front Half of Cylinder ---
    % Constant Space (Sigma) Lines - Front
    \foreach \ang in {195, 225, 255, 285, 315, 345} {
        \draw[sigmacolor, line width=0.6pt] ({\Rx*cos(\ang)}, {\Ry*sin(\ang)}) -- ({\Rx*cos(\ang)}, {\H+\Ry*sin(\ang)});
    }
    % Constant Time (Tau) Lines - Front
    \foreach \y in {0, 1, 2, 4} {
        \draw[taucolor, line width=0.6pt] (-\Rx, \y) arc[start angle=180, end angle=360, x radius=\Rx, y radius=\Ry];
    }

    % Top Surface Edge
    \draw[taucolor, line width=0.6pt] (0, \H) ellipse[x radius=\Rx, y radius=\Ry];
    % Side Edges
    \draw[thick] (-\Rx, 0) -- (-\Rx, \H);
    \draw[thick] (\Rx, 0) -- (\Rx, \H);

    % The State (Specific constant time slice) - Front
    \draw[statecolor, line width=4pt, opacity=0.3] (-\Rx, \statetau) arc[start angle=180, end angle=360, x radius=\Rx, y radius=\Ry];
    \draw[statecolor, line width=1.5pt] (-\Rx, \statetau) arc[start angle=180, end angle=360, x radius=\Rx, y radius=\Ry];

    % Coordinate Flow Arrows for Cylinder
    \draw[-Stealth, thick] (-\Rx-0.5, 0) -- (-\Rx-0.5, \H) node[midway, left=2pt] {Euclidean time $\tau$};
    \draw[-Stealth, thick] (0, \H+0.8) arc[start angle=90, end angle=-30, x radius=1.8, y radius=0.7] node[below right=-2pt] {Space $\sigma$};

    % State Pointer (Top Left)
    %\draw[-Stealth, thick, statecolor] (-\Rx-2.5, \statetau) node[left, align=center, text=black] {State $|\Psi\rangle$\\at $\tau = \text{const}$} to[out=0, in=180] (-\Rx-0.1, \statetau);


    % =========================================================
    % CENTER: CONFORMAL MAP ARROW
    % =========================================================
    \draw[-{Stealth[length=4mm, width=3.5mm]}, line width=2pt, draw=black!70] 
        (2.3, 2.5) to[out=15, in=165] 
        node[midway, above=6pt, text=black] {\large \textbf{Conformal Map}} 
        node[midway, below=4pt, text=black] {\large $z = e^{\tau + i\sigma}$} 
        (7.0, 2.5);


    % =========================================================
    % RIGHT: COMPLEX PLANE (z = r * e^{i*theta})
    % =========================================================
    
    % Axis Titles
    %\node[align=center, font=\large] at (\C, -1.8) {\textbf{Complex Plane ($\mathbb{C}$)}\\ $z = e^w$};

    % Mapped Constant Space Lines (Radial rays)
    % Exactly corresponding to the angles on the cylinder!
    \foreach \ang in {15, 45, 75, 105, 135, 165, 195, 225, 255, 285, 315, 345} {
        \draw[sigmacolor, line width=0.6pt] (\C, \Cy) -- ++(\ang:\Pr);
    }

    % Mapped Constant Time Lines (Concentric circles)
    \foreach \r in {0.5, 1.2, 2.8} {
        \draw[taucolor, line width=0.6pt] (\C, \Cy) circle (\r);
    }

    % The State (Mapped constant time slice)
    \draw[statecolor, line width=4pt, opacity=0.3] (\C, \Cy) circle (2.0);
    \draw[statecolor, line width=1.5pt] (\C, \Cy) circle (2.0);

    % Operator / Vacuum Insertion (Origin)
    \filldraw[fill=opcolor, draw=black, thick] (\C, \Cy) circle (3pt);

    % Coordinate Flow Arrows for Plane
    \draw[-Stealth, thick] (\C, \Cy) ++(30:0.4) -- ++(30:3) node[above right=-2pt, fill=white, inner sep=1pt] {Radial time $|z| = r$};
    \draw[-Stealth, thick] (\C, \Cy) ++(0:1.4) arc[start angle=0, end angle=30, radius=1.4] node[midway, right=2pt] {Space $\sigma$};

    % Operator Pointer (Bottom Right)
    \draw[-Stealth, thick, opcolor] (\C+2.5, \Cy-2) node[right, align=center, text=black] {Local Operator\\$\mathcal{O}(0)$ at $z = 0$} to[out=180, in=315] (\C+0.1, \Cy-0.1);

    % State Pointer (Top Right)
    %\draw[-Stealth, thick, statecolor] (\C+3.5, \Cy+2) node[right, align=center, text=black] {State $|\Psi\rangle$\\at circle $|z| = R$} to[out=180, in=45] (\C+1.414, \Cy+1.414);

\end{tikzpicture}
